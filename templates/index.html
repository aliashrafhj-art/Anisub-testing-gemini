ব্যাকগ্রাউন্ড থ্রেড শুরু thread = threading.Thread(target=process_engine, args=(task_id,)) thread.start() return jsonify({'task_id': task_id}) @app.route('/api/status/') def get_status(task_id): if task_id not in tasks: return jsonify({'error': 'Task not found'}), 404 return jsonify(tasks[task_id]) #
